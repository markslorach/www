## Code block

```typescript
import { allWriting } from "content-collections";

export default function Writing() {
  const posts = allWriting
    .filter((post) => post.published)
    .sort((a, b) => new Date(b.date).getTime() - new Date(a.date).getTime())
    .slice(0, 5);

  return (
    <section>
      {posts.map((post) => (
        <PostCard key={post._meta.fileName} post={post} />
      ))}
    </section>
  );
}
```
